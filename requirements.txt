# --- Backend: API server ---
fastapi                 # Web framework for the backend API
uvicorn[standard]       # Server that runs the FastAPI app
python-dotenv           # Loads API keys from the .env file
pydantic                # Data validation for API requests/responses

# --- AI services ---
google-genai            # Google's SDK for Gemini embeddings
groq                    # Groq's SDK for fast LLM answers
portkey-ai              # Optional LLM gateway — latency/cost dashboards, caching

# --- Vector database ---
qdrant-client           # Client for Qdrant Cloud vector search
flashrank               # Local cross-encoder reranker (no API needed)

# --- Document parsing ---
pypdf                   # PDF text extraction
python-docx             # Word document parsing
python-pptx             # PowerPoint parsing
beautifulsoup4          # HTML text extraction

# --- Observability ---
logfire[fastapi]                 # Pipeline tracing (spans/traces); no-op without LOGFIRE_TOKEN

# --- Reliability ---
tenacity                # Retry logic with proper wait strategies

# --- Agent orchestration ---
langgraph               # State machine: planner -> retriever? -> responder, with memory

# --- Evals ---
ragas                    # Industry-standard RAG eval metrics (faithfulness, relevancy, precision, recall)
langchain-groq           # ragas' LLM-as-judge wrapper requires a LangChain client
langchain-google-genai   # ragas' embedding-based metrics require a LangChain embeddings client

# --- Frontend ---
streamlit               # Chat UI
requests                # Lets the UI call the backend over HTTP
